1. Sistema de calificaciones detallado

Este algoritmo utiliza un bucle `for`, condicionales `if-else` y arreglos. Evalúa 10 notas, las suma, calcula el promedio y lo clasifica según el rango.

Análisis:

A continuación se describe el análisis del algoritmo

Datos de entrada:

  • Nota 1
  • Nota 2
  • Nota 3
  • Nota 4
  • Nota 5
  • Nota 6
  • Nota 7
  • Nota 8
  • Nota 9
  • Nota 10

Proceso:

Se inicializa la variable suma, se recorre un bucle del 1 al 10 donde se accede a cada input, se valida que esté entre 1.0 y 5.0, se suman y se calcula el promedio. Luego, se evalúa en qué rango se encuentra.

Datos o mensajes de salida:

Mensajes como "Felicitaciones, su promedio fue Excelente" y otros según el resultado del promedio.

Función realizada:

        
// Calcular promedio de 10 notas
function calculateAverage() {
    let suma = 0;
    let promedio = 0;
    for (let i = 1; i <= 10; i++) {
        let id = 'nota' + i;
        let input = document.getElementById(id);
        let valor = parseFloat(input.value);
        if (valor < 1.0 || valor > 5.0) {
            return alert('Por favor asegurese de que todas las notas estén entre 1.0 y 5.0');
        } else {
            suma += valor;
        }
    }
    promedio = suma / 10;
    if (promedio > 4.4) {
        alert('Felicitaciones, su promedio fue Excelente, su nota es ' + promedio);
    } else if (promedio > 3.9) {
        alert('Muy bien!, su promedio fue Sobresaliente, su nota es ' + promedio);
    } else if (promedio > 2.9) {
        alert('Hay que mejorar, su promedio fue Aceptable, su nota es ' + promedio);
    } else {
        alert('Hay que nivelar, su promedio fue Reprobado, su nota es ' + promedio);
    }
}

    Análisis:

    Datos de entrada:

  • Peso en kilogramos

  • Estatura en metros

    • Proceso:

    El usuario ingresa su peso y estatura. El algoritmo calcula el IMC utilizando la fórmula: IMC = peso / (estatura)^2. Luego, usando una estructura if-else, determina el estado de salud del usuario y da una recomendación:

    • IMC < 18.5: Bajo peso
    • IMC entre 18.5 y 24.9: Normal
    • IMC entre 25 y 29.9: Sobrepeso
    • IMC ≥ 30: Obesidad

    Datos o mensajes de salida:

    Recomendación de salud personalizada y el valor del IMC calculado.

    Función realizada:

    
// ALGORITMO DEL ÍNDICE DE MASA CORPORAL (IMC) Y RECOMENDACIONES DE SALUD
function imc() {
    let weightPerson = parseInt(document.getElementById('weight').value);
    let heightPerson = parseFloat(document.getElementById('height').value);
    let imcResult = weightPerson / heightPerson ** 2;
    let answerImc;

    if (imcResult < 18.5) {
        answerImc = 'Bajo peso - se recomienda consultar a un nutricionista';

        answerImc = 'Obesidad - Es importante buscar asesoramiento médico para un plan de salud integral';
    }

        answerImc + ' ya que tu índice de masa corporal (IMC) es de: ' + imcResult.toFixed(2);

// EJERCICIOS CON BUCLE WHILE
function counterPositiveAndNegative() {
    let number = parseInt(prompt('Ingrese el primer número'));
    let counterPositive = 0;
    let counterNegative = 0;

    while (number !== 0) {
        if (number < 0) {
            counterNegative++;
        } else if (number > 0) {
            counterPositive++;
        }
        number = parseInt(prompt('Ingrese un número'));
    }

    console.log('Positivos:', counterPositive, 'Negativos:', counterNegative);
    document.getElementById('positiveNumbers').textContent = counterPositive;
    document.getElementById('negativeNumbers').textContent = counterNegative;
}

// WHILE DE ACUMULADOR DE COMPRAS HASTA LLEGAR A 100.000

    let counterAcumulate = 0;

    while (counterAcumulate < 100000) {

    27. Carrito de compras

    Carrito de compras

    Este algoritmo permite agregar productos con nombre, precio y cantidad a un carrito. Muestra un resumen y el total acumulado.

    Análisis:

    Descripción del análisis:

    Datos de entrada:

    • Nombre del producto
    • Precio del producto
    • Cantidad del producto

    Proceso:

    Se agregan productos en un ciclo mientras el usuario quiera, almacenando cada producto en un arreglo. Se calcula subtotal y total.

    Datos o mensajes de salida:

    • Resumen con nombre, precio, cantidad y subtotal de cada producto
    • Total del carrito

    Funciones realizadas:

    
let carrito = [];

function addProductToCart() {
  let seguir = true;

  while (seguir) {
    let name = prompt('Ingrese el nombre del producto que desea agregar al carrito');
    let price = parseFloat(prompt('Ingrese el precio del producto'));
    while (isNaN(price) || price <= 0) {
      price = parseFloat(prompt('Por favor ingrese un precio válido (número positivo)'));
    }
    let cantidad = parseInt(prompt('Ingrese la cantidad de productos que desea agregar al carrito'));
    while (isNaN(cantidad) || cantidad <= 0) {
      cantidad = parseInt(prompt('Por favor ingrese una cantidad válida (número entero positivo)'));
    }

    let product = {
      nombre: name,
      precio: price,
      cantidad: cantidad
    };

    carrito.push(product);

    let answer = prompt('¿Desea agregar otro producto al carrito? (si/no)').toLowerCase();
    if (answer !== 'si') {
      seguir = false;
    }

    showCart();
  }
}

function showCart() {
  let total = 0;
  let resum = 'Resumen del carrito:\n';
  for (let i = 0; i < carrito.length; i++) {
    let product = carrito[i];
    let subtotal = product.precio * product.cantidad;
    total += subtotal;

    resum += `${product.nombre} - Precio: $${product.precio.toFixed(2)}, Cantidad: ${product.cantidad}, Subtotal: $${subtotal.toFixed(2)}\n`;
  }
  resum += `Total: $${total.toFixed(2)}`;
  alert(resum);
}

const peliculas = [
  { titulo: 'Inception', genero: 'Ciencia Ficción', año: 2010 },
  { titulo: 'Titanic', genero: 'Romance', año: 1997 },
  { titulo: 'Interstellar', genero: 'Ciencia Ficción', año: 2014 },
  { titulo: 'Coco', genero: 'Animación', año: 2017 },
];

function movies() {
    let search = prompt('Ingrese el genero de la película que desea buscar');
    let found = false;

    for (let i = 0; i < peliculas.length; i++) {
        if (search.toLowerCase() === peliculas[i].genero.toLowerCase()) {
            alert(`Película encontrada:\nTítulo: ${peliculas[i].titulo}\nGénero: ${peliculas[i].genero}\nAño: ${peliculas[i].año}`);

        }
    }
    if (!found) {

    }
}

function showMoviesOver2000() {

      mensaje += `Película: ${p.titulo}, Género: ${p.genero}, Año: ${p.año}\n`;
    });
    alert(mensaje);
  }
